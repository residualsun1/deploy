3.1 ggplot2

3.1.1 散点图 抖散图 箱线图 直方图

散点图

ggplot(data=core,aes(x=height,y=weight))+
  geom_point()

#设置颜色
ggplot(data=core,
       aes(x=height,y=weight,color=gender))+
  geom_point()

抖散图

ggplot(data=core,aes(x=educ,y=log(income)))+
  geom_point()

core %>% 
  ggplot(aes(x = educ, y =log(income))) +
  geom_jitter() #为每个点的位置增加了少量的随机变化，能够处理点相互重叠的问题

箱线图

ggplot(data=core,
       aes(x=educ,y=log(income),fill=stype,color=stype))+
  geom_boxplot(alpha=0.7)
## Warning: Removed 624 rows containing non-finite values (`stat_boxplot()`).

#添加标题、横纵轴标签

       aes(x=educ,y=log(income),fill=stype,color=stype))+
  geom_boxplot(alpha=0.7)+
  labs(x="受教育状况",y="2020年个人收入",title="受教育状况和个人收入关系分布箱线图")
